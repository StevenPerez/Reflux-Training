REACT COMPONENTS

  • connect (State)

ACTION
      
       const Actions = Reflux.createActions(["buyPizza"]);
STORE
      
        const state = {
          happy: false
        };

        const Store = Reflux.createStore({
          listenables: [Actions],
          getInitialState: function() {
            return state;
          },
          onBuyPizza: function() {
            state.happy = true;
            this.trigger(state);
          },
        });
REACT COMPONENT
      
       const HappyPizza = React.createClass({
          mixins: [
            Reflux.listenTo(Store, "onBuyPizza"),
            Reflux.connect(Store, "pizza"), // this.state.pizza.happy
          ],
          onBuyPizza(state) {
            if (state.happy)
              alert(" Yay !!! ");
            else
              alert(" :'( ");
          },
          render() {
            < Button onClick={Actions.buyPizza}>Buy Pizza< /Button>
          }
      });
